83 double LatLon::distance(const LatLon& p1, const LatLon& p2) {
84  // Spherical law of cosines
85  // The Haversine version might be better for small distances
86 
87  const double v1 = toRadians(p1.latitude());
88  const double v2 = toRadians(p2.latitude());
89  const double dl = toRadians(p2.longitude() - p1.longitude());
90  return std::acos( std::sin(v1) * std::sin(v2) + std::cos(v1) * std::cos(v2) * std::cos(dl) ) * EARTH_MEAN_RADIUS;
91 }

93 double LatLon::initialBearing(const LatLon& from, const LatLon& to) {
94  const double v1 = toRadians(from.latitude());
95  const double v2 = toRadians(to.latitude());
96  const double l1 = toRadians(from.longitude());
97  const double l2 = toRadians(to.longitude());
98 
99  const double x = std::sin(l2 - l1) * std::cos(v2);
100  const double y = std::cos(v1) * std::sin(v2) - std::sin(v1) * std::cos(v2) * std::cos(l2 - l1);
101  const double bearingRadians = std::atan2(y, x);
102  // Convert to degrees and normalize
103  const double degrees = toDegrees(bearingRadians);
104  return std::fmod(degrees + 360, 360);
105 }

107 double LatLon::finalBearing(const LatLon &from, const LatLon &to) {
108  const double initial = initialBearing(from, to);
109  return std::fmod(initial + 180, 360);
110 }
