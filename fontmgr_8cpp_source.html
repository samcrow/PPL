110 /*
111  * Helper function to figure out which textured font size
112  * would be the closest to what the client requested so that
113  * we can reuse textures.
114  *
115  * require_exact is a flag ... the client may not want rounding.
116  * If the client has a lot of fonts and zooms or scales them,
117  * rounding is a non-issue. But for a main UI font at fixed resolution
118  * exact resolution looks a lot better.
119  *
120  */
121 static int FindNearestFontSize(int inSize, bool require_exact)
122 {
123  // If we want an exact size, well, nothing to do!
124  if (require_exact) return inSize;
125 
126  // What this does is it finds the nearest font size
127  // that is an interval of 10. It rounds up by adding half of
128  // the interval we're looking for and then it basically subtracts
129  // out a value to bring it to the nearest interval. The end result
130  // is a round and truncate.
131  int rounded = (inSize + (FM_FONT_SIZE_INTERVAL/2));
132  int nearest = rounded - (rounded % FM_FONT_SIZE_INTERVAL);
133 
134  // Enfore at least the smallest interval
135  if(nearest < FM_FONT_SIZE_INTERVAL)
136  nearest = FM_FONT_SIZE_INTERVAL;
137 
138  return nearest;
139 }

168 FontHandle FontMgr::loadFont(const char* inFontPath, const char * inStartMem, const char * inEndMem, unsigned int inSizePx, bool require_exact)
169 {
170  // First see if we've used this font before
171  if(tex_map_.count(inFontPath))
172  {
173  // If we have, now let's see if we've already loaded a size that's "close enough"
174  // for the user. Right now we only do font sizes in intervals of 10 and it'll round
175  // to the nearest 10th. So if the user asks for a 13px and we have a 10px, he'll get
176  // the 10px. If he doesn't have a 10px, he'll make his own.
177  for(TextureMap_t::iterator it = tex_map_.begin(); it != tex_map_.end(); ++it)
178  {
179  if((*it).second->tex_font_size == (FindNearestFontSize(inSizePx, require_exact)) &&
180  strcmp((*it).first.c_str(), inFontPath) == 0)
181  {
182  // Increment our reference count
183  (*it).second->nRefCnt++;
184  return (*it).second;
185  }
186  }
187  }
188 
189  // We need to make them a texture but first we have to round their size to a 10px interval
190  inSizePx = FindNearestFontSize(inSizePx, require_exact);
191 
192  FT_Face face;
193  FontHandle info = new FontInfo_t;
194  FT_GlyphSlot glyph;
195 
196  if (inStartMem)
197  {
198  int test = !FT_New_Memory_Face(library_, (const FT_Byte*) inStartMem, inEndMem - inStartMem, FM_DEFAULT_FACE_INDEX, &face);
199  assert(test);
200  (void)test;
201  }
202  else
203  {
204  int test = !FT_New_Face(library_, inFontPath, FM_DEFAULT_FACE_INDEX, &face);
205  assert(test);
206  (void)test;
207  }
208  int test = !FT_Set_Char_Size(face, 0, inSizePx * 64, FM_DEVICE_RES_H, FM_DEVICE_RES_V);
209  assert(test);
210  (void)test;
211 
212  CalcTexSize(&face, &info->tex_height, &info->tex_width);
213 
214  unsigned char* textureData = new unsigned char[info->tex_height * info->tex_width];
215  // Create the texture number that we'll be tied to
216 #ifdef BUILD_FOR_STANDALONE
217  glGenTextures(1, (GLuint*)&info->tex_id);
218  // Now we bind to it
219  glBindTexture(GL_TEXTURE_2D, info->tex_id);
220 #else
221  XPLMGenerateTextureNumbers(&info->tex_id, 1);
222  XPLMSetGraphicsState(0,1,0,0,1,0,0);
223  XPLMBindTexture2d(info->tex_id, 0);
224 #endif
225  // We have to 0 out the memory or we'll get artifacts when the glyphs are cut
226  memset(textureData, 0, info->tex_height * info->tex_width);
227 
228  if(info->tex_id == 0)
229  {
230  delete[] textureData;
231  return 0;
232  }
233 
234  info->tex_font_size = inSizePx;
235 
236  float maxHeight = 0;
237  float x_off = FM_PIX_PADDING / 2.0, y_off = FM_PIX_PADDING / 2.0;
238 
239  // PHILIPP SEZ: The range of ARINC739 characters starts 4 characters before the
240  // ASCII range, and they are needed in X-Plane to display special stuff on the CDU.
241  for(int n = FM_BASE_ASCII_RANGE-4; n <= FM_PEAK_ASCII_RANGE; ++n)
242  {
243  // If there's no glyph for this font, we have nothing to do
244  if(FT_Load_Char(face, n, FT_LOAD_RENDER))
245  continue;
246 
247  // Merely a shortcut from typing this out everytime
248  glyph = face->glyph;
249 
250  float width = glyph->metrics.width / 64.0;
251  float height = glyph->metrics.height / 64.0;
252 
253  int test = !(glyph->metrics.width % 64);
254  assert(test);
255  (void)test;
256  test = !(glyph->metrics.height % 64);
257  assert(test);
258 
259  if(height > maxHeight)
260  maxHeight = height;
261 
262  if(x_off + width + FM_PIX_PADDING > info->tex_width)
263  {
264  x_off = 0;
265  if(y_off + maxHeight > info->tex_height)
266  y_off = 0;
267  else
268  y_off += (maxHeight+FM_PIX_PADDING);
269  }
270 
271  // Set all of the coordinates of the glyph
272  info->char_left_x[n] = x_off;
273  info->char_right_x[n] = x_off + width;
274  info->char_top_y[n] = y_off + height;
275  info->char_bot_y[n] = y_off;
276  info->dx[n] = glyph->metrics.horiBearingX;
277  info->dy[n] = glyph->metrics.horiBearingY - glyph->metrics.height;
278  info->advance[n] = glyph->metrics.horiAdvance;
279 
280  // Copy the glyph bitmap into temporary memory.
281  CopyBitmapSection(height, width, 0, 0, glyph->bitmap.buffer,
282  info->tex_height, info->tex_width, x_off, y_off, textureData);
283 
284 
285  if(x_off + width + FM_PIX_PADDING<= info->tex_width)
286  x_off += width + FM_PIX_PADDING;
287  else
288  {
289  if(y_off + maxHeight < info->tex_height)
290  y_off += (maxHeight+FM_PIX_PADDING);
291  else
292  y_off = maxHeight;
293 
294  x_off = width + FM_PIX_PADDING;
295  }
296  }
297 
298  // By setting the lineheight, we can determine how tall we can stretch
299  // our fonts later on give a certain height constraint.
300  info->line_height = (float) inSizePx * (float) face->height / (float) face->units_per_EM;
301  info->line_descent = (float) inSizePx * (float) -face->descender / (float) face->units_per_EM;
302  info->line_ascent = (float) inSizePx * (float) face->ascender / (float) face->units_per_EM;
303  // Set the reference count
304  info->nRefCnt = 1;
305 
306  tex_map_.insert(TextureMap_t::value_type(inFontPath, info));
307 
308  // Create some texture memory in OpenGL for this font
309  //OGL_ERROR(glTexImage2D(GL_TEXTURE_2D, 0, GL_ALPHA, info->tex_width,
310  // info->tex_height, 0, GL_ALPHA, GL_UNSIGNED_BYTE, textureData))
311 
312  // Now build mipmaps based on this texture
313  char buf[256];
314  sprintf(buf, "Trying to build mipmaps for font %s, tex width %d, tex height %d, texture data %p\n", inFontPath, info->tex_width, info->tex_height, textureData);
315  XPLMDebugString(buf);
316  OGL_ERROR(gluBuild2DMipmaps(GL_TEXTURE_2D, GL_ALPHA, info->tex_width, info->tex_height, GL_ALPHA, GL_UNSIGNED_BYTE, textureData))
317 
318  // Ben sez: use nearest neighbor for exact-size fonts...pixel accurate!
319  // Use linear for scaled fonts....less artifacts when we scale.
320  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, require_exact ? GL_NEAREST : GL_LINEAR);
321  /*int error = */glGetError();
322  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, require_exact ? GL_NEAREST : GL_LINEAR_MIPMAP_LINEAR);
323  /*error = */glGetError();
324 
325  delete[] textureData;
326 
327  FT_Done_Face(face);
328  return info;
329 }

331 void FontMgr::unloadFont(FontHandle inFont)
332 {
333  if(!inFont)
334  return;
335 
336  // Decrement our reference count.
337  // If we're the last to use this font
338  // then we cleanup the resources.
339  if(--(inFont->nRefCnt) > 0)
340  return;
341 
342  // Delete the texture from OpenGL memory
343  OGL_ERROR(glDeleteTextures(1, (GLuint*)&(inFont->tex_id)))
344 
345  // Remove it from our map
346  for(TextureMap_t::iterator it = tex_map_.begin(); it != tex_map_.end(); ++it)
347  {
348  if(inFont == (*it).second)
349  {
350  tex_map_.erase(it);
351  break;
352  }
353  }
354 
355  // Delete the struct's memory
356  delete inFont;
357 }

359 void FontMgr::CalcTexSize(
360  FT_Face* inFace,
361  int* outHeight,
362  int* outWidth)
363 {
364  if(!outHeight || !outWidth)
365  return;
366 
367  int curRowWidth = 0, maxHeight = 0, nRows = 1;
368 
369  // We iterate through every letter that we care about
370  // and record the tallest letter and the width of all
371  // letters combined as well as the widest letter.
372  for(int n = FM_BASE_ASCII_RANGE; n <= FM_PEAK_ASCII_RANGE; ++n)
373  {
374  if(FT_Load_Char(*inFace, n, FT_LOAD_RENDER))
375  continue;
376 
377  // If the bitmap doesn't exist, we have nothing to do
378  if(!(*inFace)->glyph->bitmap.rows || !(*inFace)->glyph->bitmap.width)
379  continue;
380 
381  int width = (*inFace)->glyph->metrics.width / 64.0;
382  int height = (*inFace)->glyph->metrics.height / 64.0;
383 
384  // If any of the sizes are 0, we have nothing to do
385  if(!(*inFace)->glyph->metrics.width || !(*inFace)->glyph->metrics.height)
386  continue;
387 
388  if(height > maxHeight)
389  maxHeight = height;
390 
391  // Add this to our sum with some room for padding as well
392  curRowWidth += width + FM_PIX_PADDING;
393 
394  // If it doesn't fit, we add a new row and put it there
395  if(curRowWidth > FM_DEFAULT_TEX_WIDTH)
396  {
397  nRows++;
398  curRowWidth = width + FM_PIX_PADDING;
399  }
400  }
401 
402  // We can assume our width is 1024px right now. This can be inefficient
403  // but for our purposes, I doubt it'll be a problem because our textures
404  // are fairly cheap resource-wise.
405  *outWidth = FM_DEFAULT_TEX_WIDTH;
406 
407  // Our height is how many rows we need times the height of each row
408  *outHeight = nRows * (maxHeight+FM_PIX_PADDING);
409 
410  // Now round our height up to the nearest power of 2
411  int roundedHeight = 0;
412  for(int n = 0; n < 11; ++n)
413  {
414  if(roundedHeight > *outHeight)
415  break;
416 
417  roundedHeight = 1L << n;
418  }
419 
420  *outHeight = roundedHeight;
421 }

469 void FontMgr::drawRange(
470  FontHandle inFont,
471  float color[4], // 4-part color, featuring alpha.
472  float inLeft,
473  float inBottom,
474  float inRight,
475  float inTop,
476  const char * inStart,
477  const char * inEnd,
478  FontAlign_t fontAlign)
479 {
480  if(!inFont || !inStart)
481  return;
482  float l, b, r, t, scale;
483 
484 #ifdef BUILD_FOR_STANDALONE
485  glBindTexture(GL_TEXTURE_2D, inFont->tex_id);
486 #else
487  XPLMSetGraphicsState(0,1,0,0,1,0,0);
488  XPLMBindTexture2d(inFont->tex_id, 0);
489 #endif
490 
491  // Determine how much to scale the font to make it right the height
492  scale = (inTop - inBottom) / inFont->line_height;
493 
494  if (fontAlign == ALIGN_LEFT)
495  {
496  // Left align is the simple case - just start on the left. We don't need inRIght and we don't pre-measure.
497  l = inLeft;
498  }
499  else
500  {
501  // Justification cases. First we measure the string and see how it compares to the box. Set up left as needed.
502  float width, slop;
503  width = measureRange(inFont, inTop - inBottom, inStart, inEnd);
504  // Slop is the amount of room we have horizontally between how much we need
505  // and how much space they've given up. A negative number means we'll go out
506  // of our bounds. This is a dummy and not used ifi we're left aligned.
507  slop = inRight - inLeft - width;
508  if(fontAlign == ALIGN_RIGHT)
509  l = inLeft + slop;
510  else
511  // NOTE: Due to a possible driver bug, if slop is an even number (which has a remainder
512  // when divded by 2 in an ALIGN_CENTER) the font gets skewed by 1 pixel. So we hack
513  // around it by rounding down and accepting the 0.5pixel error.
514  l = inLeft + floorf((slop / 2.0));
515  }
516 
517  b = floorf(inBottom + inFont->line_descent * scale);
518 
519  // Set the color and let's start drawing
520  glColor4fv(color);
521  glBegin(GL_QUADS);
522 
523  for (const char * p = inStart; p < inEnd; ++p)
524  {
525  float s1 = (inFont->char_left_x[static_cast<int>(*p)] - (FM_PIX_PADDING / 2.0)) / (float)inFont->tex_width;
526  float s2 = (inFont->char_right_x[static_cast<int>(*p)] + (FM_PIX_PADDING / 2.0)) / (float)inFont->tex_width;
527  float t1 = (inFont->char_bot_y[static_cast<int>(*p)] - (FM_PIX_PADDING / 2.0)) / (float)inFont->tex_height;
528  float t2 = (inFont->char_top_y[static_cast<int>(*p)] + (FM_PIX_PADDING / 2.0)) / (float)inFont->tex_height;
529  // Our top is the bottom plus the height of the Glyph and adjusted for scale;
530  t = b + (inFont->char_top_y[static_cast<int>(*p)] - inFont->char_bot_y[static_cast<int>(*p)]) * scale;
531  // Our right side is the left plus the width of the Glyph and adjusted for scale
532  r = l + (inFont->char_right_x[static_cast<int>(*p)] - inFont->char_left_x[static_cast<int>(*p)]) * scale;
533 
534  // Now we adjust the positioning of the letter in both the X and Y directions
535  // remembering to scale the adjustment properly
536  l += ((inFont->dx[static_cast<int>(*p)] / 64.0) - (FM_PIX_PADDING / 2.0)) * scale;
537  r += ((inFont->dx[static_cast<int>(*p)] / 64.0) + (FM_PIX_PADDING / 2.0)) * scale;
538  b += ((inFont->dy[static_cast<int>(*p)] / 64.0) - (FM_PIX_PADDING / 2.0)) * scale;
539  t += ((inFont->dy[static_cast<int>(*p)] / 64.0) + (FM_PIX_PADDING / 2.0)) * scale;
540 
541  // Splat the character down where it belongs now
542  glTexCoord2f(s1,t2); glVertex2f(l, b);
543  glTexCoord2f(s1,t1); glVertex2f(l, t);
544  glTexCoord2f(s2,t1); glVertex2f(r, t);
545  glTexCoord2f(s2,t2); glVertex2f(r, b);
546 
547  // Now we have to undo our dx for the next time around
548  l -= ((inFont->dx[static_cast<int>(*p)] / 64.0) - (FM_PIX_PADDING / 2.0)) * scale;
549  r -= ((inFont->dx[static_cast<int>(*p)] / 64.0) + (FM_PIX_PADDING / 2.0)) * scale;
550  b -= ((inFont->dy[static_cast<int>(*p)] / 64.0) - (FM_PIX_PADDING / 2.0)) * scale;
551  t -= ((inFont->dy[static_cast<int>(*p)] / 64.0) + (FM_PIX_PADDING / 2.0)) * scale;
552 
553  // Now we set our l position for the next time...advance already
554  // includes the width of the font to "advance" it for next time.
555  l += ((inFont->advance[static_cast<int>(*p)] / 64.0) * scale);
556  }
557  glEnd();
558 
559  // Restore graphics state by disabling texturing
560 #ifndef BUILD_FOR_STANDALONE
561  XPLMSetGraphicsState(0,0,0,0,1,0,0);
562 #endif
563 
564 }

574 float FontMgr::measureRange(
575  FontHandle inFont,
576  float inHeight,
577  const char * inStart,
578  const char * inEnd)
579 {
580  float scale = inHeight / inFont->line_height;
581  float width = 0;
582 
583  for (const char * p = inStart; p < inEnd; ++p)
584  {
585  width += ((inFont->advance[static_cast<int>(*p)] / 64.0) * scale) ;
