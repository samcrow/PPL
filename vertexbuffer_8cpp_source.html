104 void VertexBuffer::setupForDraw()
105 {
106  if (num_nrm_)
107  glEnableClientState(GL_NORMAL_ARRAY);
108  if (num_col_)
109  glEnableClientState(GL_COLOR_ARRAY);
110  if (num_tx_ || num_tx2_)
111  glEnableClientState(GL_TEXTURE_COORD_ARRAY);
112  glBindBuffer(GL_ARRAY_BUFFER, vbo_);
113 
114  float* p = NULL;
115 
116  int stride = stride_floats_*sizeof(GLfloat);
117 
118  glVertexPointer(num_vrt_, GL_FLOAT, stride, p);
119  p += num_vrt_;
120  if (num_nrm_)
121  glNormalPointer(GL_FLOAT, stride, p);
122  p += num_nrm_;
123  if(num_tx_)
124  glTexCoordPointer(num_tx_, GL_FLOAT, stride, p);
125  p += num_tx_;
126  if (num_tx2_)
127  {
128  glClientActiveTexture(GL_TEXTURE1);
129  glTexCoordPointer(num_tx2_, GL_FLOAT, stride, p);
130  glClientActiveTexture(GL_TEXTURE0);
131  }
132  p += num_tx2_;
133  if(num_col_)
134  glColorPointer(num_col_, GL_FLOAT, stride, p);
135 }
