41 std::streamsize AptDatCache::findAirport(const std::string& code) {
42  // First check if the airport is already cached
43  try {
44  return airportCache.at(code);
45  }
46  catch(std::out_of_range&) {
47  if(reader.allAirportsRead()) {
48  // Don't have anything more to read
49  throw NoSuchAirportException("No airport with the requested ID was found");
50  }
51  // Read from the file
52  if(!reader.readInProgress()) {
53  // Block
54  findAirport(code);
55  try {
56  return airportCache.at(code);
57  }
58  catch(...) {
59  throw NoSuchAirportException("No airport with the requested ID was found");
60  }
61  }
62  else {
63  // Reading is in progress; Add this promise to be handled by the other thread
64  throw ReadInProgressException("Reading in progress; cannot start. Please wait.");
65  }
66  }
67 }
