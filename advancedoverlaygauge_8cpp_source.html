67 XPLMCursorStatus AdvancedOverlayGauge::handle2dCursorCallback(XPLMWindowID /*window_id*/, int x, int y)
68 {
69  if(isVisible()) {
70  // The provided X and Y values are in screen coordinates from the bottom left corner
71  // Transform X and Y by the lower left corner of the window to make them relative to the window
72  x -= getLeft2d();
73  y -= getBottom2d();
74 
75  mouseX = x;
76  mouseY = y;
77  // Iterate backwards over the click regions, from last to first added
78  for(auto it = regions.rbegin(); it != regions.rend(); it++) {
79  // Stop if a region has been found
80  bool matched = (**it).isInRegion(x, y);
81  if(matched) {
82  customCursor = (**it).cursorType();
83  if(customCursor == CursorType::Default) {
84  return xplm_CursorArrow;
85  }
86  else {
87  // Hide the cursor
88  return xplm_CursorHidden;
89  }
90  }
91  }
92  }
93  // No click region; use default cursor
94  customCursor = CursorType::Default;
95  return xplm_CursorArrow;
96 }
