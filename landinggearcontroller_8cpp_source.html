69 // Flight loop callback
70 float LandingGearController::callback(float, float, int)
71 {
72 
73  // Get deploy ratio from the model
74  float deployRatio = model().deployRatio();
75 
76  // See if deploy ratio has changed
77  if(std::isnan(lastDeployRatio)) {
78  // Deploy ratio not yet set. Don't do any of the following conditions
79  }
80  else if(isBasicallyOne(deployRatio) && !isBasicallyOne(lastDeployRatio)) {
81  // Just finished deploying
82  extendedSignal();
83  }
84  else if(isBasicallyZero(deployRatio) && !isBasicallyZero(lastDeployRatio)) {
85  // Just finished retracting
86  retractedSignal();
87  }
88  else if(!isBasicallyZero(deployRatio) && isBasicallyZero(lastDeployRatio)) {
89  // Just started extending
90  extensionStartedSignal();
91  }
92  else if(!isBasicallyOne(deployRatio) && isBasicallyOne(lastDeployRatio)) {
93  // Just started retracting
94  retractionStartedSignal();
95  }
96 
97  lastDeployRatio = deployRatio;
98 
99  return -1;
100 }
