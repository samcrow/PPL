37 ALSoundBuffer::ALSoundBuffer(const std::string& filename) throw(SoundPlayingError):
38  m_name(filename)
39 {
40  ALfloat source_position[] = { 0.0, 0.0, 0.0 };
41  ALfloat source_velocity[] = { 0.0, 0.0, 0.0 };
42  m_loop = AL_FALSE;
43  // Clear present errors
44  alGetError();
45  m_buffer = LoadWav(filename);

47  if( m_buffer == AL_NONE)

60  {
61  throw SoundSourceError("Error in creating source");
62  }
63  alSourcei (m_source, AL_BUFFER, m_buffer );
64  alSourcef (m_source, AL_PITCH, 1.0 );

66  alSourcefv(m_source, AL_POSITION, source_position);

68  alSourcei (m_source, AL_LOOPING, m_loop );
69  if (alGetError() != AL_NO_ERROR)

88 bool ALSoundBuffer::play() throw(SoundPlayingError)
89 {
90  if (!(alIsSource( m_source ) == AL_TRUE))
91  {
92  std::stringstream stream;
93  stream << "Error: " << m_name << " has no source";
94  throw SoundSourceError(stream.str());
95  }
96  ALint buf;
97  alGetSourcei(m_source, AL_BUFFER, &buf);
98  if (!(alIsBuffer(buf) == AL_TRUE))
99  {
100  std::stringstream stream;
101  stream << "Error: " << m_name << " has no buffer";
102  throw SoundBufferError(stream.str());
103  }
104 
105  if (alGetError() != AL_NO_ERROR)
106  {
107  std::stringstream stream;
108  stream << "Error, cannot play " << m_name;
109  throw SoundPlayingError(stream.str());
110  }
111  ALfloat listener_position[]= { 0.0, 0.0, 0.0 };
112  ALfloat listener_velocity[] = { 0.0, 0.0, 0.0 };
113  ALfloat listener_orientation[] = { 0.0, 0.0, -1.0, 0.0, 1.0, 0.0 };
114  alListenerfv(AL_POSITION, listener_position);
115  alListenerfv(AL_VELOCITY, listener_velocity);
116  alListenerfv(AL_ORIENTATION, listener_orientation);
117  if (alGetError() != AL_NO_ERROR)
118  {
119  std::stringstream stream;
120  stream << "Error cannot play " << m_name <<
121  ". Setup of source and listener failed";
122  throw SoundPlayingError(stream.str());
123  }
124  alSourcePlay( m_source );
125  return true;
126 }

164 static uint16_t readByte16(const unsigned char buffer[2]) {
165 #if APL && defined(__ppc__)
166  return (buffer[0] << 8) + buffer[1];
167 #else
168  return (buffer[1] << 8) + buffer[0];
169 #endif
170 }
171 static uint32_t readByte32(const unsigned char buffer[4]) {
172 #if APL && defined(__ppc__)
173  return (buffer[0] << 24) + (buffer[1] << 16) + (buffer[2] << 8) + buffer[3];
174 #else
175  return (buffer[3] << 24) + (buffer[2] << 16) + (buffer[1] << 8) + buffer[0];
176 #endif
177 }

184 // - Alut source code: static BufferData *loadWavFile (InputStream *stream)
185 // http://www.openal.org/repos/openal/tags/freealut_1_1_0/alut/alut/src/alutLoader.c
186 ALuint PPL::LoadWav(const std::string& fileName) {
187  const unsigned int BUFFER_SIZE = 32768; // 32 KB buffers
188  long bytes;
189  std::vector<char> data;
190  ALenum format;
191  ALsizei freq;
192 
193  // Local resources
194  FILE *f = NULL;
195  char *array = NULL;
196  ALuint buffer = AL_NONE;
197 
198  alGetError();
199 
200  // Main process
201  try {
202 
203  // Open for binary reading
204  f = fopen(fileName.c_str(), "rb");
205  if (!f)
206  throw PPL::ALSoundBuffer::SoundBufferError("LoadWav: Could not load wav from " + fileName);
207 
208  // buffers
209  char magic[5];
210  magic[4] = '\0';
211  unsigned char buffer32[4];
212  unsigned char buffer16[2];
213 
214  // check magic
215  if(fread(magic,4,1,f) != 1)
216  throw PPL::ALSoundBuffer::SoundBufferError("LoadWav: Cannot read wav file "+ fileName );
217  if(std::string(magic) != "RIFF")
218  throw ALSoundBuffer::SoundBufferError("LoadWav: Wrong wav file format. This file is not a .wav file (no RIFF magic): "+ fileName );
219 
220  // skip 4 bytes (file size)
221  fseek(f,4,SEEK_CUR);
222 
223  // check file format
224  if(fread(magic,4,1,f) != 1)
225  throw ALSoundBuffer::SoundBufferError("LoadWav: Cannot read wav file "+ fileName );
226  if(std::string(magic) != "WAVE")
227  throw ALSoundBuffer::SoundBufferError("LoadWav: Wrong wav file format. This file is not a .wav file (no WAVE format): "+ fileName );
228 
229  // check 'fmt ' sub chunk (1)
230  if(fread(magic,4,1,f) != 1)
231  throw ALSoundBuffer::SoundBufferError("LoadWav: Cannot read wav file "+ fileName );
232  if(std::string(magic) != "fmt ")
233  throw ALSoundBuffer::SoundBufferError("LoadWav: Wrong wav file format. This file is not a .wav file (no 'fmt ' subchunk): "+ fileName );
234 
235  // read (1)'s size
236  if(fread(buffer32,4,1,f) != 1)
237  throw ALSoundBuffer::SoundBufferError("LoadWav: Cannot read wav file "+ fileName );
238  uint32_t subChunk1Size = readByte32(buffer32);
239  if (subChunk1Size < 16)
240  throw ALSoundBuffer::SoundBufferError("Wrong wav file format. This file is not a .wav file ('fmt ' chunk too small, truncated file?): "+ fileName );
241 
242  // check PCM audio format
243  if(fread(buffer16,2,1,f) != 1)
244  throw ALSoundBuffer::SoundBufferError("LoadWav: Cannot read wav file "+ fileName );
245  uint16_t audioFormat = readByte16(buffer16);
246  if (audioFormat != 1)
247  throw ALSoundBuffer::SoundBufferError("LoadWav: Wrong wav file format. This file is not a .wav file (audio format is not PCM): "+ fileName );
248 
249  // read number of channels
250  if(fread(buffer16,2,1,f) != 1)
251  throw ALSoundBuffer::SoundBufferError("LoadWav: Cannot read wav file "+ fileName );
252  uint16_t channels = readByte16(buffer16);
253 
254  // read frequency (sample rate)
255  if (fread(buffer32,4,1,f) != 1)
256  throw ALSoundBuffer::SoundBufferError("LoadWav: Cannot read wav file "+ fileName );
257  uint32_t frequency = readByte32(buffer32);
258 
259  // skip 6 bytes (Byte rate (4), Block align (2))
260  fseek(f,6,SEEK_CUR);
261 
262  // read bits per sample
263  if(fread(buffer16,2,1,f) != 1)
264  throw ALSoundBuffer::SoundBufferError("LoadWav: Cannot read wav file "+ fileName );
265  uint16_t bps = readByte16(buffer16);
266 
267  if (channels == 1)
268  format = (bps == 8) ? AL_FORMAT_MONO8 : AL_FORMAT_MONO16;
269  else
270  format = (bps == 8) ? AL_FORMAT_STEREO8 : AL_FORMAT_STEREO16;
271 
272  // check 'data' sub chunk (2)
273  if(fread(magic,4,1,f) != 1)
274  throw ALSoundBuffer::SoundBufferError("LoadWav: Cannot read wav file "+ fileName );
275  if(std::string(magic) != "data")
276  throw ALSoundBuffer::SoundBufferError("LoadWav: Wrong wav file format. This file is not a .wav file (no data subchunk): "+ fileName );
277 
278  if(fread(buffer32,4,1,f) != 1)
279  throw ALSoundBuffer::SoundBufferError("LoadWav: Cannot read wav file "+ fileName );
280  uint32_t subChunk2Size = readByte32(buffer32);
281 
282  // The frequency of the sampling rate
283  freq = frequency;
284  if (sizeof(freq) != sizeof(frequency))
285  throw ALSoundBuffer::SoundBufferError("LoadWav: freq and frequency different sizes");
286 
287  array = new char[BUFFER_SIZE];
288 
289  while (data.size() != subChunk2Size) {
290  // Read up to a buffer's worth of decoded sound data
291  bytes = fread(array, 1, BUFFER_SIZE, f);
292 
293  if (bytes <= 0)
294  break;
295 
296  if (data.size() + bytes > subChunk2Size)
297  bytes = subChunk2Size - data.size();
298 
299  // Append to end of buffer
300  data.insert(data.end(), array, array + bytes);
301  };
302 
303  delete []array;
304  array = NULL;
305 
306  fclose(f);
307  f = NULL;
308 
309  alGenBuffers(1, &buffer);
310  if(alGetError() != AL_NO_ERROR)
311  throw ALSoundBuffer::SoundBufferError("LoadWav: Could not generate buffer");
312  if(AL_NONE == buffer)
313  throw ALSoundBuffer::SoundBufferError("LoadWav: Could not generate buffer");
314 
315  alBufferData(buffer, format, &data[0], data.size(), freq);
316  if(alGetError() != AL_NO_ERROR)
317  throw ALSoundBuffer::SoundBufferError("LoadWav: Could not load buffer data");
318 
319  return buffer;
320  } catch (std::runtime_error& er) {
321  if (buffer)
322  if (alIsBuffer(buffer) == AL_TRUE)
323  alDeleteBuffers(1, &buffer);
324 
325  if (array)
326  delete []array;
327 
328  if (f)
329  fclose(f);
330  throw;
331  }
332 }
