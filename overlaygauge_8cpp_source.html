279 void OverlayGauge::draw2dWindowCallback(XPLMWindowID)
280 {
281  alpha_ = instrumentBrightness();
282  if (wantRedraw())
283  {
284  GLint xp_fbo;
285  glGetIntegerv(GL_FRAMEBUFFER_BINDING, &xp_fbo);
286  // set rendering destination to FBO
287  glBindFramebufferEXT(GL_FRAMEBUFFER_EXT, fbo_);
288  glPushAttrib(GL_VIEWPORT_BIT);
289  glViewport(0,0,width_3d_, height_3d_);
290  glMatrixMode (GL_PROJECTION);
291  glPushMatrix();
292  glLoadIdentity();
293  glOrtho(0,width_3d_, 0, height_3d_, 0, 1);
294  glMatrixMode (GL_MODELVIEW);
295  glPushMatrix();
296  glLoadIdentity();
297 
298  // clear buffers
299  if (wantClearTexture())
300  glClearColor(0,0,0,0);
301  else
302  glClearColor(0,0,0,1);
303  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
304 
305  // draw a scene to a texture directly
306  if (xplane_version_ > 10000)
307  glDisableClientState(GL_VERTEX_ARRAY);
308  draw(0, height_3d_, width_3d_, 0);
309  if (xplane_version_ > 10000)
310  glEnableClientState(GL_VERTEX_ARRAY);
311 
312  // unbind FBO
313 
314  glPopMatrix();
315  glMatrixMode (GL_PROJECTION);
316  glPopMatrix();
317  glMatrixMode(GL_MODELVIEW);
318  glPopAttrib();
319  glBindFramebufferEXT(GL_FRAMEBUFFER_EXT, xp_fbo);
320  }
321 
322  if (visible_2d_)
323  {
324  int left, top, right, bottom;
325  XPLMGetWindowGeometry(window2d_id_, &left, &top, &right, &bottom);
326 
327  if (frameIsBackground())
328  drawFrameTexture(left, top, right, bottom);
329  drawTexture(gauge_texture_, left+frame_off_x_, top-frame_off_y_, left+frame_off_x_+width_view_3d_, top -frame_off_y_-height_view_3d_, alpha_, (frameIsBackground() || wantClearTexture()));
330  if (!frameIsBackground())
331  drawFrameTexture(left, top, right, bottom);
332 
333  if (window_has_keyboard_focus_)
334  {
335  static float color[] = { 1.f, 0.5f, 0.f};

361 int OverlayGauge::handle2dClickCallback(XPLMWindowID window_id, int x, int y, XPLMMouseStatus mouse)
362 {
363  //printf("mouse (%d,%d)\n",x,y);
364  static int dX = 0, dY = 0;
365  static int Weight = 0, Height = 0;
366  int Left, Top, Right, Bottom;
367 
368  if (!visible_2d_)
369  return 0;
370 
372  XPLMGetWindowGeometry(window_id, &Left, &Top, &Right, &Bottom);
373  int x_rel = x - Left;
374  int y_rel = y - Bottom;
375  switch(mouse) {
376  case xplm_MouseDown:
378  if (coordInRect(x, y, Left, Top, Left+50, Top-50))
379  {
380  if (allow_keyboard_grab_)
381  {
382  if (window_has_keyboard_focus_)
383  {
384  XPLMTakeKeyboardFocus(0);
385  window_has_keyboard_focus_ = false;
386  } else
387  {
388  XPLMTakeKeyboardFocus(window_id);
389  window_has_keyboard_focus_ = true;
390  }
391  }
392  }
393  if (coordInRect(x, y, Left+50, Top, Right-50, Top-50))
394  {
395  dX = x - Left;
396  dY = y - Top;
397  Weight = Right - Left;
398  Height = Bottom - Top;
399  window_is_dragging_ = true;
400  }
401  if (coordInRect(x, y, Right-50, Top, Right, Top-50))
402  {
403  XPLMTakeKeyboardFocus(0);
404  window_has_keyboard_focus_ = false;
405  setVisible(false);
406  }
407  if (coordInRect(x, y, Left, Top-50, Right, Bottom))
408  {
409  handleNonDragClick(x_rel, y_rel);
410  }
411  break;
412  case xplm_MouseDrag:
414  if (window_is_dragging_)
415  {
416  Left = (x - dX);
417  Right = Left + Weight;
418  Top = (y - dY);
419  Bottom = Top + Height;
420  XPLMSetWindowGeometry(window_id, Left, Top, Right, Bottom);
421  }
422  break;
423  case xplm_MouseUp:
424  window_is_dragging_ = false;
425  if (coordInRect(x, y, Left, Top-50, Right, Bottom))
426  {
427  handleNonDragClickRelease(x_rel, y_rel);
428  }
429  break;
430  }
431  return 1;
432 }
