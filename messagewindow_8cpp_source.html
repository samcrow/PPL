120 void MessageWindow::display()
121 {
122  createSurroundingBox();
123  createInnerScreen();
124  // break up string in portions
126  const unsigned int max_chars = 70;
127  std::list<std::string> list_of_words;
128  splitStr(list_of_words, m_message, " ", false);
129 
130  std::list<std::string> list_of_sequences;
131  std::string seq;
132  while (!list_of_words.empty())
133  {
134  if (seq.size() + list_of_words.front().size() < max_chars)
135  {
136  // Adding the current word to the sequence doesn't exceed max length. Append it and remove it from words
137  seq.append(list_of_words.front()).append(" ");
138  list_of_words.pop_front();
139  } else {
140  // The string would grow too large. save its actual contents and clear the temporary string
141  list_of_sequences.push_back(seq);
142  seq.clear();
143  }
144  }
145  if (!seq.empty())
146  list_of_sequences.push_back(seq);
147  // for each sequence of maximum length, create a caption inside the screen widget
148  m_caption_widgets_list.reserve(list_of_sequences.size());
149  for (int i=0 ; !list_of_sequences.empty() ; i++)
150  {
151  m_caption_widgets_list.push_back( XPCreateWidget(m_left + 10,
152  m_top - 30 - i * 12,
153  m_right - 10,
154  m_top - 40 - i * 12,
155  1,
156  list_of_sequences.front().c_str(),
157  0,
158  m_box_widget,
159  xpWidgetClass_Caption)
160  );
161  list_of_sequences.pop_front();
162  }
163 
164  XPSetWidgetProperty(m_screen_widget, xpProperty_SubWindowType, xpSubWindowStyle_SubWindow);
165  XPSetWidgetProperty(m_box_widget, xpProperty_Object, reinterpret_cast<long>(this));
166  XPAddWidgetCallback(m_box_widget, widgetCallback);
167  m_is_displayed = true;
168 }

215 int MessageWindow::splitStr(std::list<std::string>& L, const std::string& seq, const std::string& _1cdelim, bool _removews )
216 {
217  typedef std::string::size_type ST;
218  std::string delims = _1cdelim;
219  std::string STR;
220  if(delims.empty()) delims = "\n\r";
221  if(_removews) delims += " ";
222 
223  ST pos=0, LEN = seq.size();
224  while(pos < LEN )
225  {
226  STR=""; // Init/clear the STR token buffer
227 
228  // remove any delimiters including optional (white)spaces
229  while( (delims.find(seq[pos]) != std::string::npos) && (pos < LEN) ) ++pos;
230 
231  // leave if @eos
232  if(pos==LEN) return L.size();
233  // Save token data
234  while( (delims.find(seq[pos]) == std::string::npos) && (pos < LEN) ) STR += seq[pos++];
235  // put valid STR buffer into the supplied list
236  if( ! STR.empty() ) L.push_back(STR);
237  }
238  return L.size();
239 }
